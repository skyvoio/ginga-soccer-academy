1. The Main Layout & Routing (src/App.tsx)
This handles the global "shell" of the website with the sticky glass-morphism navbar and the specific HashRouter for GitHub Pages compatibility.

TypeScript
import React, { useState } from 'react';
import { HashRouter, Routes, Route, Link, useLocation } from 'react-router-dom';
import { Menu, X, Trophy, User, ChevronRight } from 'lucide-react';
import BookingWizard from './BookingWizard';

// --- Components ---

const Navbar = () => {
  const [isOpen, setIsOpen] = useState(false);
  const location = useLocation();

  const navLinks = [
    { name: 'PROGRAMS', path: '/programs' },
    { name: 'GINGAMAX', path: '/gingamax' },
    { name: 'SCHEDULE', path: '/schedule' },
    { name: 'MEMBERS', path: '/booking' }, // The Booking Wizard
  ];

  const isActive = (path: string) => location.pathname === path;

  return (
    <nav className="fixed w-full z-50 top-0 bg-neutral-950/80 backdrop-blur-md border-b border-white/10">
      <div className="max-w-7xl mx-auto px-6 h-20 flex items-center justify-between">
        {/* Logo Area */}
        <Link to="/" className="flex items-center gap-2 group">
          <div className="h-8 w-8 bg-amber-500 rounded-sm rotate-45 group-hover:rotate-0 transition-transform duration-500" />
          <span className="text-2xl font-black tracking-tighter text-white uppercase italic">
            Ginga<span className="text-amber-500">.</span>
          </span>
        </Link>

        {/* Desktop Nav */}
        <div className="hidden md:flex items-center gap-8">
          {navLinks.map((link) => (
            <Link
              key={link.name}
              to={link.path}
              className={`text-sm font-bold tracking-widest hover:text-amber-500 transition-colors ${
                isActive(link.path) ? 'text-amber-500' : 'text-neutral-400'
              }`}
            >
              {link.name}
            </Link>
          ))}
          <Link 
            to="/login"
            className="flex items-center gap-2 px-6 py-2 border border-white/20 hover:bg-white hover:text-black transition-all duration-300 uppercase text-xs font-bold tracking-widest"
          >
            <User size={14} />
            Login
          </Link>
        </div>

        {/* Mobile Toggle */}
        <button onClick={() => setIsOpen(!isOpen)} className="md:hidden text-white">
          {isOpen ? <X /> : <Menu />}
        </button>
      </div>

      {/* Mobile Menu */}
      {isOpen && (
        <div className="md:hidden bg-neutral-950 absolute w-full h-screen border-t border-white/10 p-6 flex flex-col gap-6">
          {navLinks.map((link) => (
            <Link
              key={link.name}
              to={link.path}
              onClick={() => setIsOpen(false)}
              className="text-2xl font-black text-white uppercase"
            >
              {link.name}
            </Link>
          ))}
        </div>
      )}
    </nav>
  );
};

2. The Booking Wizard (src/BookingWizard.tsx)
This is the complex logic component. It uses framer-motion for that luxury slide effect and implements the download/payment gate logic.

TypeScript
import React, { useState } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { Check, Download, CreditCard, ChevronLeft, Calendar } from 'lucide-react';

// --- Types ---
type ProgramCategory = 'TRAINING' | 'CAMPS' | 'RENTALS';

interface Program {
  id: string;
  category: ProgramCategory;
  title: string;
  price: string;
  details: string;
  isCamp: boolean;
}

// --- Data ---
const PROGRAMS: Program[] = [
  { id: 'p1', category: 'TRAINING', title: 'Private Session', price: '$175 + tax', details: '120 Min | 1-on-1 Coaching', isCamp: false },
  { id: 'p2', category: 'TRAINING', title: 'Group Session', price: '$50 + tax', details: '90 Min | Small Group Tactics', isCamp: false },
  { id: 'p3', category: 'TRAINING', title: 'Just Play', price: '$50 + tax', details: '120 Min | Sundays Only', isCamp: false },
  { id: 'p4', category: 'TRAINING', title: 'GingaMax Speed', price: '$50 + tax', details: 'Powered by Maximus', isCamp: false },
  { id: 'c1', category: 'CAMPS', title: 'March Break Camp', price: '$500', details: 'March 11-15 | Intensive', isCamp: true },
  { id: 'c2', category: 'CAMPS', title: 'Summer Camp', price: '$500/wk', details: 'July - August', isCamp: true },
  { id: 'r1', category: 'RENTALS', title: 'Full Turf Rental', price: '$150/hr', details: 'Full Pitch Access', isCamp: false },
];

// --- Components ---

const StepIndicator = ({ current, total }: { current: number; total: number }) => (
  <div className="flex gap-2 mb-8">
    {Array.from({ length: total }).map((_, i) => (
      <div
        key={i}
        className={`h-1 flex-1 transition-all duration-500 ${
          i + 1 <= current ? 'bg-amber-500' : 'bg-neutral-800'
        }`}
      />
    ))}
  </div>
);

const BookingWizard = () => {
  const [step, setStep] = useState(1);
  const [selectedProgram, setSelectedProgram] = useState<Program | null>(null);
  const [hasDownloaded, setHasDownloaded] = useState(false);

  // Animation variants
  const variants = {
    enter: { opacity: 0, x: 20 },
    center: { opacity: 1, x: 0 },
    exit: { opacity: 0, x: -20 },
  };

  const handleProgramSelect = (program: Program) => {
    setSelectedProgram(program);
    setStep(2); // Move to validation
    // Simulate instant validation for UX
    setTimeout(() => setStep(3), 600);
  };

  const handleDownload = () => {
    // Simulate download delay
    setTimeout(() => setHasDownloaded(true), 1000);
  };

  return (
    <div className="min-h-screen pt-32 pb-20 px-6 max-w-5xl mx-auto">
      <div className="mb-10">
        <h2 className="text-4xl md:text-5xl font-black uppercase italic tracking-tighter text-white">
          Reserve Your <span className="text-amber-500">Spot</span>
        </h2>
        <p className="text-neutral-400 mt-2 font-mono text-sm">SECURE YOUR PLACE ON THE PITCH.</p>
      </div>

      <StepIndicator current={step} total={4} />

      <AnimatePresence mode="wait">
        
        {/* STEP 1: SELECTION */}
        {step === 1 && (
          <motion.div
            key="step1"
            variants={variants}
            initial="enter"
            animate="center"
            exit="exit"
            transition={{ duration: 0.4 }}
            className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4"
          >
            {PROGRAMS.map((program) => (
              <button
                key={program.id}
                onClick={() => handleProgramSelect(program)}
                className="group relative bg-neutral-900 border border-neutral-800 p-8 text-left hover:border-amber-500 transition-all duration-300"
              >
                <div className="absolute top-0 right-0 p-2 opacity-0 group-hover:opacity-100 transition-opacity">
                  <div className="bg-amber-500 text-black p-1"><ChevronRight size={16} /></div>
                </div>
                <span className="text-xs font-bold text-neutral-500 mb-2 block">{program.category}</span>
                <h3 className="text-xl font-bold uppercase text-white mb-1 group-hover:text-amber-500 transition-colors">
                  {program.title}
                </h3>
                <p className="text-2xl font-light text-white mb-4">{program.price}</p>
                <div className="border-t border-neutral-800 pt-4 mt-4">
                  <p className="text-xs text-neutral-400 font-mono flex items-center gap-2">
                    <Calendar size={12} /> {program.details}
                  </p>
                </div>
              </button>
            ))}
          </motion.div>
        )}

        {/* STEP 2: VALIDATING (Interstitial) */}
        {step === 2 && (
          <motion.div
            key="step2"
            variants={variants}
            initial="enter"
            animate="center"
            exit="exit"
            className="flex flex-col items-center justify-center py-20"
          >
            <div className="w-12 h-12 border-4 border-neutral-800 border-t-amber-500 rounded-full animate-spin mb-6" />
            <p className="font-mono text-sm text-neutral-400">CHECKING AVAILABILITY...</p>
          </motion.div>
        )}

        {/* STEP 3 & 4: REGISTRATION & PAYMENT */}
        {(step === 3 || step === 4) && selectedProgram && (
          <motion.div
            key="step3"
            variants={variants}
            initial="enter"
            animate="center"
            exit="exit"
            className="max-w-2xl mx-auto"
          >
            <button 
              onClick={() => setStep(1)} 
              className="mb-6 text-neutral-500 hover:text-white flex items-center gap-2 text-sm font-bold uppercase tracking-widest"
            >
              <ChevronLeft size={16} /> Back to Selection
            </button>

            <div className="bg-neutral-900 border border-neutral-800 p-8 md:p-12">
              <div className="flex justify-between items-start mb-8 border-b border-neutral-800 pb-8">
                <div>
                  <h3 className="text-2xl font-bold uppercase text-white">{selectedProgram.title}</h3>
                  <p className="text-amber-500 font-mono mt-1">{selectedProgram.price}</p>
                </div>
                <div className="text-right hidden md:block">
                  <p className="text-xs text-neutral-500 font-mono">CONFIRMATION ID</p>
                  <p className="text-sm text-white">REQ-{Math.floor(Math.random() * 9000) + 1000}</p>
                </div>
              </div>

              {/* Step 3: Download Logic */}
              <div className="mb-8">
                <h4 className="text-sm font-bold text-neutral-400 uppercase tracking-widest mb-4">
                  Step 1: Registration Dossier
                </h4>
                <div className={`p-6 border transition-colors duration-300 ${hasDownloaded ? 'border-green-500/30 bg-green-500/5' : 'border-neutral-700 bg-neutral-950'}`}>
                  <div className="flex items-center justify-between">
                    <div>
                      <p className="text-white font-bold">
                        {selectedProgram.isCamp ? '2026 Camp Application' : 'Academy Registration Form'}
                      </p>
                      <p className="text-xs text-neutral-500 mt-1">PDF Document • 1.2 MB</p>
                    </div>
                    
                    {!hasDownloaded ? (
                      <button 
                        onClick={handleDownload}
                        className="bg-white text-black px-4 py-2 text-xs font-bold uppercase hover:bg-amber-500 transition-colors flex items-center gap-2"
                      >
                        <Download size={14} /> Download
                      </button>
                    ) : (
                      <div className="flex items-center gap-2 text-green-500 text-xs font-bold uppercase">
                        <Check size={14} /> Downloaded
                      </div>
                    )}
                  </div>
                  <p className="text-xs text-neutral-500 mt-4 italic">
                    *Please complete and return to info@gingasoccer.ca after payment.
                  </p>
                </div>
              </div>

              {/* Step 4: Payment Logic (Reveals after download) */}
              <div className={`transition-all duration-500 ${hasDownloaded ? 'opacity-100 translate-y-0' : 'opacity-30 translate-y-4 pointer-events-none'}`}>
                 <h4 className="text-sm font-bold text-neutral-400 uppercase tracking-widest mb-4">
                  Step 2: Finalize Placement
                </h4>
                <button 
                  className="w-full bg-gradient-to-r from-amber-500 to-amber-600 text-black font-bold uppercase tracking-widest py-4 hover:shadow-[0_0_20px_rgba(245,176,65,0.3)] transition-all flex items-center justify-center gap-3"
                >
                  <CreditCard size={18} /> Proceed to Checkout
                </button>
              </div>

            </div>
          </motion.div>
        )}
      </AnimatePresence>
    </div>
  );
};

export default BookingWizard;
